 40    """
 41    BOExperiment is a class designed to facilitate Bayesian Optimization experiments using the [Ax platform](https://ax.dev/).
 42    It encapsulates the experiment setup, including features, outcomes, constraints, and optimization methods.
 43
 44    Example
 45    -------
 46    ```python
 47    features, outcomes = read_experimental_data('data.csv', out_pos=[-2, -1])
 48    experiment = BOExperiment(features, 
 49                              outcomes, 
 50                              N=5, 
 51                              maximize={'out1':True, 'out2':False}
 52                              )
 53    experiment.suggest_next_trials()
 54    experiment.plot_model(metricname='outcome1')
 55    experiment.plot_model(metricname='outcome2', linear=True)
 56    experiment.plot_model(metricname='outcome1', slice_values={'feature1': 5})
 57    experiment.plot_optimization_trace()
 58    experiment.plot_pareto_frontier()
 59    experiment.get_best_parameters()
 60    experiment.update_experiment({'feature1': [4]}, {'outcome1': [0.4]})
 61    experiment.plot_model()
 62    experiment.plot_optimization_trace()
 63    experiment.plot_pareto_frontier()
 64    experiment.get_best_parameters()
 65    ```

128    @features.setter
129    def features(self, value):
130        """
131        Set the features of the experiment with validation.
132        """
133        if not isinstance(value, dict):
134            raise ValueError("features must be a dictionary")
135        self._features = value
136        for name in self._features.keys():
137            if self.ranges and name in self.ranges.keys():
138                self._features[name]['range'] = self.ranges[name]
139            else:
140                if self._features[name]['type'] == 'text':
141                    self._features[name]['range'] = list(set(self._features[name]['data']))
142                elif self._features[name]['type'] == 'int':
143                    self._features[name]['range'] = [int(np.min(self._features[name]['data'])),
144                                                     int(np.max(self._features[name]['data']))]
145                elif self._features[name]['type'] == 'float':
146                    self._features[name]['range'] = [float(np.min(self._features[name]['data'])),
147                                                     float(np.max(self._features[name]['data']))]
148        if self._first_initialization_done:
149            self.initialize_ax_client()

407    def initialize_ax_client(self):
408        """
409        Initialize the AxClient with the experiment's parameters, objectives, and constraints.
410        """
411        print('\n========   INITIALIZING MODEL   ========\n')
412        self.ax_client = AxClient(verbose_logging=False, suppress_storage_errors=True)
413        self.parameters = []
414        for name, info in self._features.items():
415            if info['type'] == 'text':
416                self.parameters.append({
417                    "name": name,
418                    "type": "choice",
419                    "values": [str(val) for val in info['range']],
420                    "value_type": "str"})
421            elif info['type'] == 'int':
422                self.parameters.append({
423                    "name": name,
424                    "type": "range",
425                    "bounds": [int(np.min(info['range'])),
426                               int(np.max(info['range']))],
427                    "value_type": "int"})
428            elif info['type'] == 'float':
429                self.parameters.append({
430                    "name": name,
431                    "type": "range",
432                    "bounds": [float(np.min(info['range'])),
433                               float(np.max(info['range']))],
434                    "value_type": "float"})
435
436        self.ax_client.create_experiment(
437            name="bayesian_optimization",
438            parameters=self.parameters,
439            objectives={self.out_names[i]:
440                ObjectiveProperties(minimize=not self._maximize[self.out_names[i]])
441                    for i in range(len(self.out_names))},
442            parameter_constraints=self._feature_constraints,
443            outcome_constraints=self._outcome_constraints,
444            overwrite_existing_experiment=True
445        )
446
447        if len(next(iter(self._outcomes.values()))['data']) > 0:
448            for i in range(len(next(iter(self._outcomes.values()))['data'])):
449                params = {name: info['data'][i] for name, info in self._features.items()}
450                outcomes = {name: info['data'][i] for name, info in self._outcomes.items()}
451                self.ax_client.attach_trial(params)
452                self.ax_client.complete_trial(trial_index=i, raw_data=outcomes)
453
454        self.set_model()
455        self.set_gs()

467    def set_gs(self):
468        """
469        Set the generation strategy for the experiment.
470        This method is called after initializing the AxClient.
471        """
472        self.clear_trials()
473        if self._optim == 'bo':
474            if not self.model:
475                self.set_model()
476            self.gs = GenerationStrategy(
477                steps=[GenerationStep(
478                            model=Models.BOTORCH_MODULAR,
479                            num_trials=-1,  # No limitation on how many trials should be produced from this step
480                            max_parallelism=3,  # Parallelism limit for this step, often lower than for Sobol
481                        )
482                    ]
483                )
484        elif self._optim == 'sobol':
485            self.gs = GenerationStrategy(
486                steps=[GenerationStep(
487                            model=Models.SOBOL,
488                            num_trials=-1,  # How many trials should be produced from this generation step
489                            should_deduplicate=True,  # Deduplicate the trials
490                            # model_kwargs={"seed": 165478},  # Any kwargs you want passed into the model
491                            model_gen_kwargs={},  # Any kwargs you want passed to `modelbridge.gen`
492                        )
493                    ]
494                )
495        self.generator_run = self.gs.gen(
496                experiment=self.ax_client.experiment,  # Ax `Experiment`, for which to generate new candidates
497                data=None,  # Ax `Data` to use for model training, optional.
498                n=self._N,  # Number of candidate arms to produce
499                fixed_features=self._fixed_features, 
500                pending_observations=get_pending_observation_features(
501                    self.ax_client.experiment
502                ),  # Points that should not be re-generated
503            )

515    def suggest_next_trials(self, with_predicted=True):
516        """
517        Suggest the next set of trials based on the current model and optimization strategy.
518
519        Returns
520        -------
521
522        pd.DataFrame: 
523            DataFrame containing the suggested trials and their predicted outcomes.
524        """
525        self.clear_trials()
526        if self.ax_client is None:
527            self.initialize_ax_client()
528        if self._N == 1:
529            self.candidate = self.ax_client.experiment.new_trial(self.generator_run)
530        else:
531            self.candidate = self.ax_client.experiment.new_batch_trial(self.generator_run)
532        trials = self.ax_client.get_trials_data_frame()
533        trials = trials[trials['trial_status'] == 'CANDIDATE']
534        trials = trials[[name for name in self.names]]
535        if with_predicted:
536            topred = [trials.iloc[i].to_dict() for i in range(len(trials))]
537            preds = pd.DataFrame(self.predict(topred))
538            # add 'predicted_' to the names of the pred dataframe
539            preds.columns = [f'Predicted_{col}' for col in preds.columns]
540            preds = preds.reset_index(drop=True)
541            trials = trials.reset_index(drop=True)
542            return pd.concat([trials, preds], axis=1)
543        else:
544            return trials

546    def predict(self, params):
547        """
548        Predict the outcomes for a given set of parameters using the current model.
549
550        Parameters
551        ----------
552
553        params : List[Dict[str, Any]]
554            List of parameter dictionaries for which to predict outcomes.

556        Returns

565        f, _ = self.model.predict(obs_feats)

568    def update_experiment(self, params, outcomes):
569        """
570        Update the experiment with new parameters and outcomes, and reinitialize the AxClient.
571
572        Parameters
573        ----------
574
575        params : Dict[str, Any]
576            Dictionary of new parameters to update the experiment with.
577
578        outcomes : Dict[str, Any]
579            Dictionary of new outcomes to update the experiment with.
580        """
581        # append new data to the features and outcomes dictionaries
582        for k, v in zip(params.keys(), params.values()):
583            if k not in self._features:
584                raise ValueError(f"Parameter '{k}' not found in features")
585            if isinstance(v, np.ndarray):
586                v = v.tolist()
587            if not isinstance(v, list):
588                v = [v]
589            self._features[k]['data'] += v
590        for k, v in zip(outcomes.keys(), outcomes.values()):
591            if k not in self._outcomes:
592                raise ValueError(f"Outcome '{k}' not found in outcomes")
593            if isinstance(v, np.ndarray):
594                v = v.tolist()
595            if not isinstance(v, list):
596                v = [v]
597            self._outcomes[k]['data'] += v
598        self.initialize_ax_client()

600    def plot_model(self, metricname=None, slice_values={}, linear=False):
601        """
602        Plot the model's predictions for the experiment's parameters and outcomes.
603
604        Parameters
605        ----------
606
607        metricname : Optional[str]
608            The name of the metric to plot. If None, the first outcome metric is used.
609
610        slice_values : Optional[Dict[str, Any]]
611            Dictionary of slice values for plotting.
612
613        linear : bool
614            Whether to plot a linear slice plot. Default is False.
615
616        Returns
617        -------
618
619        plotly.graph_objects.Figure: 
620            Plotly figure of the model's predictions.
621        """
622        if self.ax_client is None:
623            self.initialize_ax_client()
624            self.suggest_next_trials()
625
626        cand_name = 'Candidate' if self._N == 1 else 'Candidates'
627        mname = self.out_names[0] if metricname is None else metricname
628        
629        param_name = [name for name in self.names if name not in slice_values.keys()]
630        par_numeric = [name for name in param_name if self._features[name]['type'] in ['int', 'float']]
631        if len(par_numeric)==1:
632            fig = plot_slice(
633                    model=self.model,
634                    metric_name=mname,
635                    density=100,
636                    param_name=par_numeric[0],
637                    generator_runs_dict={cand_name: self.generator_run},
638                    slice_values=slice_values
639                    )
640        elif len(par_numeric)==2:
641            fig = plot_contour(
642                    model=self.model,
643                    metric_name=mname,
644                    param_x=par_numeric[0],
645                    param_y=par_numeric[1],
646                    generator_runs_dict={cand_name: self.generator_run},
647                    slice_values=slice_values
648                    )
649        else:
650            fig = interact_contour(
651                    model=self.model,
652                    generator_runs_dict={cand_name: self.generator_run},
653                    metric_name=mname,
654                    slice_values=slice_values,
655                )
656
657        # Turn the figure into a plotly figure
658        plotly_fig = go.Figure(fig.data)
659
660        # Modify only the "In-sample" markers
661        trials = self.ax_client.get_trials_data_frame()
662        trials = trials[trials['trial_status'] == 'CANDIDATE']

868def flatten_dict(d, parent_key="", sep="_"):
869    """
870    Flatten a nested dictionary.
871    """
872    items = []
873    for k, v in d.items():
874        new_key = f"{parent_key}{sep}{k}" if parent_key else k
875        if isinstance(v, dict):
876            items.extend(flatten_dict(v, new_key, sep=sep).items())
877        else:
878            items.append((new_key, v))
879    return dict(items)

883def ordered_dict_to_dataframe(data):
884    """
885    Convert an OrderedDict with arbitrary nesting to a DataFrame.
886    """
887    dflat = flatten_dict(data)
888    out = []
889
890    for key, value in dflat.items():
891        main_dict = value[0]
892        sub_dict = value[1][0]
893        out.append([value for value in main_dict.values()] +
894                   [value for value in sub_dict.values()])
895
896    df = pd.DataFrame(out, columns=[key for key in main_dict.keys()] +
897                                   [key for key in sub_dict.keys()])
898    return df

902def read_experimental_data(file_path: str, out_pos=[-1]) -> (Dict[str, Dict[str, Any]], Dict[str, Dict[str, Any]]):
903    """
904    Read experimental data from a CSV file and format it into features and outcomes dictionaries.
905
906    Parameters
907    ----------
908    file_path (str) 
909        Path to the CSV file containing experimental data.
910    out_pos (list of int)
911        Column indices of the outcome variables. Default is the last column.
912
913    Returns
914    -------
915    Tuple[Dict[str, Dict[str, Any]], Dict[str, Dict[str, Any]]]
916        Formatted features and outcomes dictionaries.
917    """
918    data = pd.read_csv(file_path)
919    data = clean_names(data, remove_special=True, case_type='preserve')
920    outcome_column_name = data.columns[out_pos]
921    features = data.loc[:, ~data.columns.isin(outcome_column_name)].copy()
922    outcomes = data[outcome_column_name].copy()
923
924    feature_definitions = {}
925    for column in features.columns:
926        if features[column].dtype == 'object':
927            unique_values = features[column].unique()
928            feature_definitions[column] = {'type': 'text',
929                                           'range': unique_values.tolist()}
930        elif features[column].dtype in ['int64', 'float64']:
931            min_val = features[column].min()
932            max_val = features[column].max()
933            feature_type = 'int' if features[column].dtype == 'int64' else 'float'
934            feature_definitions[column] = {'type': feature_type,
935                                           'range': [min_val, max_val]}
936
937    formatted_features = {name: {'type': info['type'],
938                                 'data': features[name].tolist(),
939                                 'range': info['range']}
940                          for name, info in feature_definitions.items()}
941    # same for outcomes with just type and data
942    outcome_definitions = {}
943    for column in outcomes.columns:
944        if outcomes[column].dtype == 'object':
945            unique_values = outcomes[column].unique()
946            outcome_definitions[column] = {'type': 'text',
947                                           'data': unique_values.tolist()}
948        elif outcomes[column].dtype in ['int64', 'float64']:
949            min_val = outcomes[column].min()
950            max_val = outcomes[column].max()
951            outcome_type = 'int' if outcomes[column].dtype == 'int64' else 'float'
952            outcome_definitions[column] = {'type': outcome_type,
953                                           'data': outcomes[column].tolist()}
954    formatted_outcomes = {name: {'type': info['type'],
955                                 'data': outcomes[name].tolist()}
956                           for name, info in outcome_definitions.items()}
957    return formatted_features, formatted_outcomes
