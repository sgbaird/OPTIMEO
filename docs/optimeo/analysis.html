optimeo.analysis

The analysis module provides tools for data analysis and regression modeling. The main workhorse is the DataAnalysis class, which allows for encoding categorical variables, performing regression analysis, and visualizing results.

It supports both linear regression using the statsmodels package and machine learning models from sklearn. The class also provides methods for plotting Q-Q plots, box plots, histograms, and scatter plots. It includes functionality for bootstrap resampling to estimate the variability of model coefficients. The DataAnalysis class is designed to be flexible and extensible, allowing users to customize the regression analysis process.

 482    def plot_pairplot_plotly(self):
 483        """
 484        Plot a pairplot for the data using plotly.
 485
 486        Returns
 487        -------
 488        fig : plotly.graph_objs.Figure
 489            The plotly figure.
 490        """
 491        
 492        # Get the column names
 493        columns = self._data.columns
 494        n_cols = len(columns)
 495
 496        # Create subplot grid with partially shared axes
 497        fig = make_subplots(
 498            rows=n_cols,
 499            cols=n_cols,
 500            shared_xaxes=True,  # We'll manually set this
 501            shared_yaxes=True,  # We'll manually set this
 502            vertical_spacing=0.05,  # Increase spacing to avoid overlap
 503            horizontal_spacing=0.05  # Increase spacing to avoid overlap
 504        )
 505
 506        # Add scatter plots and regression lines to each subplot
 507        for i in range(n_cols):
 508            for j in range(n_cols):
 509                x_data = self._data[columns[j]]
 510                y_data = self._data[columns[i]]
 511
 512                if i == j:  # Diagonal: Add KDE plot
 513                    # Calculate KDE
 514                    data = self._data[columns[i]].dropna()
 515                    if len(data) > 1:  # Ensure enough data points
 516                        kde_x = np.linspace(data.min(), data.max(), 100)
 517                        kde = stats.gaussian_kde(data)
 518                        kde_y = kde(kde_x)
 519                        kde_y = kde_y / kde_y.max() * np.max(y_data.dropna())  # Scale to match y-axis
 520
 521                        # Add KDE plot to diagonal
 522                        fig.add_trace(
 523                            go.Scatter(
 524                                x=kde_x,
 525                                y=kde_y,
 526                                mode='lines',
 527                                fill='tozeroy',
 528                                line=dict(color='rgba(29, 81, 189, 1)'),
 529                                showlegend=False
 530                            ),
 531                            row=i+1, col=j+1
 532                        )
 533                        # Ensure the y-axis is independent for diagonal plots
 534                        fig.update_yaxes(
 535                            matches=None,  # Ensure independent y-axis
 536                            showticklabels=True,  # Show y-axis values
 537                            row=i+1,
 538                            col=j+1
 539                        )
 540                else:  # Off-diagonal: Add scatter plot with regression line
 541                    # Add scatter plot
 542                    fig.add_trace(
 543                        go.Scatter(
 544                            x=x_data,
 545                            y=y_data,
 546                            mode='markers',
 547                            marker=dict(
 548                                color='rgba(29, 81, 189, 0.5)',  # Light blue with alpha=0.1
 549                                size=5
 550                            ),
 551                            showlegend=False
 552                        ),
 553                        row=i+1, col=j+1
 554                    )
 555
 556                    # Calculate and add regression line
 557                    if len(x_data.dropna()) > 1 and len(y_data.dropna()) > 1:
 558                        # Drop NaN values for regression calculation
 559                        valid_indices = x_data.notna() & y_data.notna()
 560                        x_clean = x_data[valid_indices]
 561                        y_clean = y_data[valid_indices]
 562
 563                        if len(x_clean) > 1:  # Ensure we have enough points for regression
 564                            # Calculate regression parameters
 565                            slope, intercept, r_value, p_value, std_err = stats.linregress(x_clean, y_clean)
 566                            x_range = np.linspace(x_clean.min(), x_clean.max(), 100)
 567                            y_pred = slope * x_range + intercept
 568
 569                            # Standard error of estimate
 570                            y_fit = slope * x_clean + intercept
 571                            residuals = y_clean - y_fit
 572                            dof = len(x_clean) - 2
 573                            residual_std_error = np.sqrt(np.sum(residuals**2) / dof)
 574
 575                            mean_x = np.mean(x_clean)
 576                            t_val = t.ppf(0.975, dof)  # 95% confidence
 577
 578                            se_line = residual_std_error * np.sqrt(1/len(x_clean) + (x_range - mean_x)**2 / np.sum((x_clean - mean_x)**2))
 579                            y_upper = y_pred + t_val * se_line
 580                            y_lower = y_pred - t_val * se_line
 581
 582                            # Add regression line
 583                            fig.add_trace(
 584                                go.Scatter(
 585                                    x=x_range,
 586                                    y=y_pred,
 587                                    mode='lines',
 588                                    line=dict(color='red', width=2),
 589                                    showlegend=False
 590                                ),
 591                                row=i+1, col=j+1
 592                            )
 593
 594                            # Add confidence interval area
 595                            fig.add_trace(
 596                                go.Scatter(
 597                                    x=np.concatenate([x_range, x_range[::-1]]),
 598                                    y=np.concatenate([y_upper, y_lower[::-1]]),
 599                                    fill='toself',
 600                                    fillcolor='rgba(255, 0, 0, 0.2)',
 601                                    line=dict(color='rgba(255, 0, 0, 0)'),
 602                                    showlegend=False
 603                                ),
 604                                row=i+1, col=j+1
 605                            )
 606
 607        # Update layout and axis properties
 608        fig.update_layout(
 609            margin=dict(l=10, r=10, t=50, b=50),
 610            title="Pair Plot",
 611            height=600,
 612            width=600,
 613            showlegend=False,
 614            plot_bgcolor="white"
 615        )
 616
 617        # Update all axes properties first (hiding tick labels by default)
 618        fig.update_xaxes(
 619            showgrid=True,
 620            gridcolor="lightgray",
 621            zeroline=False,
 622            showline=True,         # Show axis line
 623            linewidth=1,
 624            linecolor="black",
 625            mirror=True,           # Mirror lines to form a box
 626            showticklabels=False
 627        )
 628
 629        fig.update_yaxes(
 630            showgrid=True,
 631            gridcolor="lightgray",
 632            zeroline=False,
 633            showline=True,         # Show axis line
 634            linewidth=1,
 635            linecolor="black",
 636            mirror=True,           # Mirror lines to form a box
 637            showticklabels=False
 638        )
 639
 640        # Show tick labels and titles only for bottom row and leftmost column
 641        for i, col_name in enumerate(columns):
 642            # Bottom row: show x-axis titles and tick labels
 643            fig.update_xaxes(
 644                title_text=col_name,
 645                showticklabels=True,
 646                row=n_cols,

 652                title_text=col_name,
 653                showticklabels=True,
 654                row=i+1,
 655                col=1
 656            )

 658        return fig

 660    def plot_corr(self):
 661        """
 662        Plot a correlation matrix for the data.
 663
 664        Returns
 665        -------
 666        fig : seaborn.axisgrid.PairGrid
 667            The pairplot figure.
 668        """
 669        corr_matrix = self._data.corr(method='pearson')
 670        mask = np.triu(np.ones_like(corr_matrix, dtype=bool), k=1)
 671        # Apply mask - replace upper triangle with NaN values
 672        corr_matrix_lower = corr_matrix.copy()
 673        corr_matrix_lower.values[mask] = np.nan
 674

 727    def write_equation(self, order=1, quadratic=[]):
 728        """
 729        Write R-style equation for multivariate fitting procedure using the statsmodels package.
 730
 731        Parameters
 732        ----------
 733        order : int, optional
 734            The order of the polynomial. Default is 1.
 735        quadratic : list, optional
 736            The list of quadratic factors. Default is an empty list.
 737
 738        Returns
 739        -------
 740        str
 741            The R-style equation.
 742        """
 743        myfactors = self._factors.copy()
 744        if self._dtypes is not None:
 745            if not isinstance(myfactors, list):
 746                myfactors = myfactors.tolist()  # Convert to list to allow mutable operations
 747            for i in range(len(self._factors)):
 748                if self._dtypes[self._factors[i]] == 'object':
 749                    myfactors[i] = f'C({myfactors[i]})'
 750        eqn = f'{self._response} ~ {myfactors[0]} '
 751        for factor in myfactors[1:]:
 752            eqn += f'+ {factor} '
 753        if order > 1:
 754            for i in range(len(myfactors)):
 755                for j in range(i + 1, len(myfactors)):
 756                    eqn += f'+ {myfactors[i]}:{myfactors[j]} '
 757        if order > 2:
 758            for i in range(len(myfactors)):
 759                for j in range(i + 1, len(myfactors)):
 760                    for k in range(j + 1, len(myfactors)):
 761                        eqn += f'+ {myfactors[i]}:{myfactors[j]}:{myfactors[k]} '
 762        if order > 3:
 763            for i in range(len(myfactors)):
 764                for j in range(i + 1, len(myfactors)):
 765                    for k in range(j + 1, len(myfactors)):
 766                        for l in range(k + 1, len(myfactors)):
 767                            eqn += f'+ {myfactors[i]}:{myfactors[j]}:{myfactors[k]}:{myfactors[l]} '
 768        if len(quadratic) > 0:
 769            for factor in quadratic:
 770                eqn += f'+ I({factor}**2) '
 771        self._equation = eqn
 772        return self._equation

1230        pd.DataFrame

1232        """

1235        else:
1236            X = X.copy()
1237

1239        for factor in self._factors:
1240            if factor in self._encoders:
1241                X[factor] = self._encoders[factor].transform(X[factor].astype(str))

1250            predML = self._model.predict(X)

1277def bootstrap_coefficients(mod, X, y, n_bootstrap=100, random_state=None):
1278    """
1279    Perform bootstrap resampling to estimate the variability of model coefficients.
1280
1281    Parameters
1282    ----------
1283    mod : object
1284        The machine learning model.
1285    X : pd.DataFrame
1286        The input features.
1287    y : pd.Series
1288        The target variable.
1289    n_bootstrap : int, optional
1290        The number of bootstrap samples. Default is 100.
1291    random_state : int, optional
1292        The seed for the random number generator.
1293
1294    Returns
1295    -------
1296    results : np.ndarray
1297        The bootstrapped coefficients.
1298    """
1299    np.random.seed(random_state)
1300    n_samples = X.shape[0]
1301    results = []
1302
1303    for _ in range(n_bootstrap):
1304        # Resample the data
1305        indices = np.random.choice(np.arange(n_samples), size=n_samples, replace=True)
1306        X_resample, y_resample = X.values[indices], y.values[indices]
1307
1308        # Fit the model
1309        if isinstance(mod, RidgeCV):
1310            model = RidgeCV(alphas=np.logspace(-3, 3, 10)).fit(X_resample, y_resample)
1311            results.append(model.coef_)
1312        elif isinstance(mod, ElasticNetCV):
1313            model = ElasticNetCV(alphas=np.logspace(-3, 3, 10), l1_ratio=0.5).fit(X_resample, y_resample)
1314            results.append(model.coef_)
1315        elif isinstance(mod, LinearRegression):
1316            model = LinearRegression().fit(X_resample, y_resample)
1317            results.append(model.coef_)
1318        elif isinstance(mod, RandomForestRegressor):
1319            model = RandomForestRegressor().fit(X_resample, y_resample)
1320            results.append(model.feature_importances_)
1321        elif isinstance(mod, GradientBoostingRegressor):
1322            model = GradientBoostingRegressor().fit(X_resample, y_resample)
1323            results.append(model.feature_importances_)
1324        elif isinstance(mod, GaussianProcessRegressor):
1325            # Gaussian Process does not have coefficients or feature importances
1326            model = GaussianProcessRegressor().fit(X_resample, y_resample)
1327            results.append(np.zeros(X.shape[1]))  # Placeholder for Gaussian Process
1328        else:
1329            raise ValueError(f"Unsupported model type: {mod}")
1330
1331    return np.array(results)

 728    def write_equation(self, order=1, quadratic=[]):
 729        """
 730        Write R-style equation for multivariate fitting procedure using the statsmodels package.
 731
 732        Parameters
 733        ----------
 734        order : int, optional
 735            The order of the polynomial. Default is 1.
 736        quadratic : list, optional
 737            The list of quadratic factors. Default is an empty list.
 738
 739        Returns
 740        -------
 741        str
 742            The R-style equation.
 743        """
 744        myfactors = self._factors.copy()
 745        if self._dtypes is not None:
 746            if not isinstance(myfactors, list):
 747                myfactors = myfactors.tolist()  # Convert to list to allow mutable operations
 748            for i in range(len(self._factors)):
 749                if self._dtypes[self._factors[i]] == 'object':
 750                    myfactors[i] = f'C({myfactors[i]})'
 751        eqn = f'{self._response} ~ {myfactors[0]} '
 752        for factor in myfactors[1:]:
 753            eqn += f'+ {factor} '
 754        if order > 1:
 755            for i in range(len(myfactors)):
 756                for j in range(i + 1, len(myfactors)):
 757                    eqn += f'+ {myfactors[i]}:{myfactors[j]} '
 758        if order > 2:
 759            for i in range(len(myfactors)):
 760                for j in range(i + 1, len(myfactors)):
 761                    for k in range(j + 1, len(myfactors)):
 762                        eqn += f'+ {myfactors[i]}:{myfactors[j]}:{myfactors[k]} '
 763        if order > 3:
 764            for i in range(len(myfactors)):
 765                for j in range(i + 1, len(myfactors)):
 766                    for k in range(j + 1, len(myfactors)):
 767                        for l in range(k + 1, len(myfactors)):
 768                            eqn += f'+ {myfactors[i]}:{myfactors[j]}:{myfactors[k]}:{myfactors[l]} '
 769        if len(quadratic) > 0:
 770            for factor in quadratic:
 771                eqn += f'+ I({factor}**2) '
 772        self._equation = eqn
 773        return self._equation

743        """

745        if self._dtypes is not None:
746            if not isinstance(myfactors, list):

749                if self._dtypes[self._factors[i]] == 'object':
